import useSWR from 'swr';

async function fetcher(url) {
  const res = await window.fetch(url);
  return await res.json();
}

export function useReservations() {
  const { data, error } = useSWR(`/api/get-reservations`, fetcher)

  return {
    reservations: data,
    isLoading: !error && !data,
    isError: error,
  }
}

export function useAllReservations(loggedIn) {
  const { data, error } = useSWR(loggedIn ? '/api/get-alldata' : null, fetcher)

  }
